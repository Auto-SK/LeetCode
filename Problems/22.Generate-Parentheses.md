```python
class Solution:
    def generateParenthesis(self, n: int) -> List[str]:
        def generate(n, l, r, s):
            if l == n and r == n:
                res.append(s)
                return
            if l < n:
                generate(n, l + 1, r, s + '(')
            if l > r:
                generate(n, l, r + 1, s + ')')
        res = []
        generate(n, 0, 0, '')
        return res
```
